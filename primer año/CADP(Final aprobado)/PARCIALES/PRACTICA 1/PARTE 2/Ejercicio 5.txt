ORG 1000H
CONTRA DB "4321"
ES     DB "ACCESO PERMITIDO"
NOES   DB "ACCESO DENEGADO"
BLOQA2 DB "ACCESO BLOQUEADO"
DATO   DB ?
ORG 2000H
      MOV DL,4
      MOV CL,5
      MOV BX,OFFSET DATO
OTRO: INT 6
      INC BX
      DEC DL
      CMP DL,0
      JNZ OTRO      
      MOV CH,0
      MOV AL,CONTRA
COMP: MOV AH,DATO
      CMP AL,AH
      JNZ ERROR
EXITO:MOV BX,OFFSET ES
      MOV AL,OFFSET NOES-OFFSET ES
      INT 7
      INT 0
ERROR:MOV BX,OFFSET NOES
      MOV AL,OFFSET BLOQA2-OFFSET NOES
      DEC CL
      CMP CL,0
      JZ BLOQ
      MOV DL,4
      JMP OTRO
BLOQ: MOV BX,OFFSET BLOQA2
      MOV AL,OFFSET DATO-OFFSET BLOQA2
      INT 7
      INT 0
END
